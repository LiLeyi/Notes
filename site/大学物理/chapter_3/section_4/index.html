    • у│╗у╗ЪтљЉтцќуЋїТћЙуЃГ№╝їQ<0

    С╝ауЃГТќ╣т╝Ј№╝ѕТІЊт▒Ћ№╝їУ»ЙтєЁтЈфУђЃУЎЉуЃГС╝ат»╝№╝Ѕ№╝џ

    • уЃГС╝ат»╝№╝џтѕєтГљуб░Тњъ
    • уЃГУЙљт░ё№╝џСЙІтдѓтцфжў│тљЉтю░уљЃуџёуЃГУЙљт░ё
    • т»╣ТхЂ

    тЂџтіЪтњїС╝ауЃГТў»У┐ЄуеІжЄЈ№╝їтњїУ┐ЄуеІТюЅтЁ│

    уЃГтіЏтГдуггСИђт«џтЙІ

    УАеУ┐░

    уггСИђуДЇУАеУ┐░№╝џУЃйжЄЈт«ѕТЂњСИјУйгтїќт«џтЙІ

    уггС║їуДЇУАеУ┐░№╝џуггСИђу▒╗Т░ИтіеТю║СИЇтЈ»УЃйжђаТѕљ№╝ѕуггСИђу▒╗№╝џУЃйСИЇС╗јтцќуЋїтљИТћХУЃйтіЏ№╝їУђїтљЉтцќуЋїСИЇТќГтю░тЂџтіЪ№╝Ѕ

    уггСИЅуДЇУАеУ┐░№╝џ\(\Delta E = Q - W№╝Џd E = \delta Q - \delta W\)

    QтљИуЃГСИ║ТГБ№╝їWтљЉтцќтЂџтіЪСИ║ТГБ

    жђѓућеУїЃтЏ┤

    С╗╗СйЋу│╗у╗Ъ

    СйЊуД»тіЪтњїуЃГжЄЈуџёуА«т«џ

    СйЊуД»тіЪ

    \(dW=\vec F \cdot d \vec l=pSdl=pdV\)

    тѕЎ\(W = \int_{V_1}^{V_2}dW=\int_{V_1}^{V_2}pdV\)

    ТГцСйЊуД»тіЪтЁгт╝ЈжђѓућеС║јтЄєжЮЎТђЂУ┐ЄуеІуџёС╗╗СйЋу│╗у╗Ъ

    у│╗у╗ЪтюетЄєжЮЎТђЂУ┐ЄуеІСИГт»╣тцќуЋїТЅђтЂџтіЪтцДт░ЈтЈ»УАеуц║СИ║p-VуіХТђЂтЏЙСИіТГцУ┐ЄуеІТЏ▓у║┐СИІуџёжЮбуД»

    У┐ЄуеІТќ╣уеІСИЇУЃйуљєУДБСИ║у│╗у╗ЪуџёуЅЕТђЂТќ╣уеІ

    СИцуѓ╣жЌ┤уџётіЪ№╝ѕУІЦУи»тЙёСИЇуА«т«џ№╝ЅтЈ»С╗ЦСИ║С╗╗ТёЈтђ╝

    уГЅтјІУ┐ЄуеІ№╝џ\(W_p = p (V_2 - V_1) = p \Delta V = \nu R \Delta T\)

    уГЅТИЕУ┐ЄуеІ№╝џ\(W_T =\nu R T ln \frac {V_2} {V_1}\)

    уЃГт«╣тњїуЃГжЄЈ

    уЃГжЄЈ

    СИ║У┐ЄуеІжЄЈ

    уЃГт«╣

    ТИЕт║дТ»ЈтЇЄжФў1KТЅђтљИТћХуџёуЃГжЄЈ№╝їуД░СИ║у│╗у╗ЪуџёуЃГт«╣№╝їућеCУАеуц║№╝їтЇЋСйЇJ/K№╝ѕSI№╝Ѕ

    \(C = \frac {dQ} {dT}\)

    ТЉЕт░ћуЃГт«╣№╝џу│╗у╗ЪуЅЕУ┤еуџёжЄЈСИ║1molТЌХ№╝їС╗ќуџёуЃГт«╣тЈФтЂџТЉЕт░ћуЃГт«╣\(C_m = \frac 1 \nu \frac {dQ} {dT} = \frac {dQ_m} {dT}\)сђѓтЇЋСйЇ\(J/mol\cdot K\)

    ТЉЕт░ћт«џт«╣уЃГт«╣

    у│╗у╗ЪСйЊуД»СИЇтЈўуџёУ┐ЄуеІСИГуџёуЃГт«╣

    \(C_V = (\frac {dQ} {dT})_V\)

    ТЉЕт░ћт«џт«╣уЃГт«╣\(C_{V,m}\)

    \(C_{V,m} = \frac 1 \nu (\frac {dQ} {dT})_V = \frac 1 \nu \frac {dE} {dT} = \frac i 2 R\)

    ућетѕ░С║єуГЅт«╣У┐ЄуеІdWСИ║0№╝ї\(dQ=dE\)сђѓуљєТЃ│Т░ћСйЊтєЁУЃйтбъжЄЈ\(\Delta E = \nu C_{V,m}\Delta T\)

    ТЉЕт░ћт«џтјІуЃГт«╣

    у│╗у╗ЪтјІт╝║СИЇтЈўУ┐ЄуеІСИГуџёуЃГт«╣

    \(C_p = (\frac {dQ} {dT})_p\)

    ТЉЕт░ћт«џтјІуЃГт«╣\(C_{p,m}\)

    \(dQ = dE + dW = \frac i 2 \nu R dT + p dV = \frac i 2 \nu R dT + \nu R dT (тЏаСИ║pV = \nu R T,тЇ│pdV + Vdp = \nu R dT№╝їтіауГЅтјІТЮАС╗Х№╝їdp=0) = (\frac i 2 + 1) R\)

    У┐ѕУђХтЁгт╝Ј

    \(C_{V,m} = \frac i 2 R\\ C_{p,m} = (\frac i 2 + 1) R\)

    У┐ѕУђХтЁгт╝Ј№╝џ

    \(C_{p,m} = C_{V,m} + R\)

    Т»ћуЃГт«╣Т»ћ

    Т»ћуЃГт«╣Т»ћ№╝џ

    \(\gamma = \frac {C_{p,m}} {C_{V,m}} = \frac{i + 2} i,\gamma > 1\)

    уГЅтђ╝У┐ЄуеІСИју╗ЮуЃГУ┐ЄуеІ

    уГЅт«╣У┐ЄуеІ

    \(\frac P T \in const№╝їтЇ│V = V_0\)

    тєЁУЃй№╝џ\(dE = \frac i 2 \nu R dT ,\Delta E = \nu C_{V,m}\Delta T\)

    тіЪ№╝џ0\((dW = p dV = 0)\)

    уЃГжЄЈ№╝џ\(dQ = \nu C_{V,m}dT = dE,Q = \nu C_{V,m}\Delta T = \Delta E\)

    уГЅтјІУ┐ЄуеІ

    \(\frac V T \in const№╝їтЇ│p = p_0\)

    тєЁУЃй№╝џ\(dE = \frac i 2 \nu RdT,\Delta E = \nu C_{V,m}\Delta T\)

    тіЪ№╝џ\(dW = pdV,W= p \Delta V\)

    уЃГжЄЈ№╝џ\(dQ = \nu C_{p,m}dT = dE + dW,Q = \nu C_{p,m}\Delta T = \Delta E + W\)

    уГЅТИЕУ┐ЄуеІ

    \(pV \in const\)

    тєЁУЃй№╝џ\(\Delta E = 0\)

    тіЪ№╝џ\(dW = p dV,W = \nu R T ln \frac {V_2} {V_1} = \nu RT ln \frac {p_1}{p_2}\)

    уЃГжЄЈ№╝џ\(dQ = dW, Q=W\)

    у╗ЮуЃГУ┐ЄуеІ

    у│╗у╗ЪтњїтцќуЋїТЌауЃГжЄЈС║цТЇбуџёТЮАС╗ХСИІУ┐ЏУАїуџёУ┐ЄуеІ

    • у╗ЮуЃГТЮљТќЎжџћуд╗
    • У┐ЄуеІУ┐ЏУАїтЙѕт┐Ф№╝їТЮЦСИЇтЈіС║цТЇбуЃГжЄЈ

    УЃйжЄЈтЈўтїќуЅ╣уѓ╣№╝џ\(Q = 0\)

    \(\Delta E + W = 0\)№╝їтЇ│ТюЅ\(dE + dW = 0\)№╝ї\(\nu C_{V,m}dT + pdV = 0\)

    \(pdV + Vdp = \nu R dT\)

    тЙЌ№╝џ\((C_{V,m} + R)pdV + C_{V,m}Vdp = 0\)№╝їтЇ│\(lnp + \gamma lnV = C\)

    Т│іТЮЙтЁгт╝Ј№╝џ

    • Тќ╣уеІ1№╝џ\(pV^\gamma = C_1\)
    • Тќ╣уеІ1№╝џ\(T V^{\gamma-1} = C_2\)
    • Тќ╣уеІ1№╝џ\(p^{\gamma - 1}T^{-\gamma} = C_3\)

    у╗ЮуЃГу║┐Т»ћуГЅТИЕу║┐ТЏ┤жЎА

    тЂџтіЪ№╝џ

    \(pV^\gamma = C = p_1 V_1 ^\gamma№╝їСИћp_1V_1 = p_2V_2\\ W = \int_{V_1} ^{V_2} pdV = \frac 1 {\gamma - 1} (p_1V_1 - p_2V_2)\)

    \(\Delta E = \nu C_{V,m}(T_2 - T_1),\gamma = \frac {C_{p,m}} {C_{V,m}} = 1 + \frac R {C_{V,m}}№╝ѕтЇ│C_{V,m} = \frac R {\gamma - 1}№╝Ѕ\\ \Rightarrow \Delta E = \frac {\nu R} {\gamma - 1} (T_2 - T_1) = \frac 1 {\gamma - 1} (p_2V_2 - p_1V_1) = - W\)
